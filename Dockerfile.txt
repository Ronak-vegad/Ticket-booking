# Use Go 1.25.3 as base image
FROM golang:1.25.3

# Set working directory
WORKDIR /app

# Copy go.mod and go.sum first, then download dependencies
COPY go.mod go.sum ./
RUN go mod download

# Copy the rest of your code
COPY . .

# Build the Go app
RUN go build -o app .

# Expose port 8080 (Render sets $PORT)
EXPOSE 8080

# Run your app
CMD ["./app"]
